<script setup lang="ts">
  import { ref } from 'vue';

  const open = ref(null);

  function setOpen(type) {
    open.value = type;
  }

  function close() {
    open.value = null;
  }

  function isOpen(type) {
    return open.value === type;
  }
</script>
